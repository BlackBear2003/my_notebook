```assembly
f proc
again:
	jmp again
final:
	exit
; 在proc中间的标号 互不干扰  直接冒号就不行

	ret		; 会返回到CALL 这个函数的指令的地方##
	
	
f endp

;the same as
f:
	ret
; 有弊端 

g proc

	call f	;## 
	
g endp
```
